13int AsyncTS::_getWriteFieldsContentLength()
14{
15 size_t iField;
16 int contentLen = 0;
17
18 for (iField = 0; iField < FIELDNUM_MAX; iField++)
19 {
20 if (this->_nextWriteField[iField].length() > 0)
21 {
22 contentLen = contentLen + 8 + this->_nextWriteField[iField].length(); // &fieldX=[value]
23
24 // future-proof in case ThingSpeak allows 999 fields someday
25 if (iField > 9)
26 {
27 contentLen = contentLen + 1;
28 }
29 else if (iField > 99)
30 {
31 contentLen = contentLen + 2;
32 }
33 }
34 }
35
36 if (!isnan(this->_nextWriteLatitude))
37 {
38 contentLen = contentLen + 5 + String(this->_nextWriteLatitude,6).length(); // &lat=[value]
39 }
40
41 if (!isnan(this->_nextWriteLongitude))
42 {
43 contentLen = contentLen + 6 + String(this->_nextWriteLongitude,6).length(); // &long=[value]
44 }
45
46 if (!isnan(this->_nextWriteElevation))
47 {
48 contentLen = contentLen + 11 + String(this->_nextWriteElevation,6).length(); // &elevation=[value]
49 }
50
51 if (this->_nextWriteStatus.length() > 0)
52 {
53 contentLen = contentLen + 8 + this->_nextWriteStatus.length(); // &status=[value]
54 }
55
56 if (this->_nextWriteTwitter.length() > 0)
57 {
58 contentLen = contentLen + 9 + this->_nextWriteTwitter.length(); // &twitter=[value]
59 }
60
61 if (this->_nextWriteTweet.length() > 0)
62 {
63 contentLen = contentLen + 7 + this->_nextWriteTweet.length(); // &tweet=[value]
64 }
65
66 if (this->_nextWriteCreatedAt.length() > 0)
67 {
68 contentLen = contentLen + 12 + this->_nextWriteCreatedAt.length(); // &created_at=[value]
69 }
70
71 if (contentLen == 0)
72 {
73 return 0;
74 }
75
76 contentLen = contentLen + 13; // add 14 for '&headers=false', subtract 1 for missing first '&'
77
78 return contentLen;
79}

82{
83 // Supported range is -999999000000 to 999999000000

202String AsyncTS::_getJSONValueByKey(String textToSearch, String key)
203{
204 if (textToSearch.length() == 0)
205 {
206 return String("");
207 }
208
209 String searchPhrase = String("\"");

211 searchPhrase.concat("\":\"");
212

214
215 if (fromPosition == -1)
216 {
217 // return because there is no status or it's null
218 return String("");
219 }
220
221 fromPosition = fromPosition + searchPhrase.length();
222
223 int toPosition = textToSearch.indexOf("\"", fromPosition);
224

226 {

228 return String("");

231 textToSearch.remove(toPosition);

269unsigned int AsyncTS::_send()
270{
271 if (_request.available() == 0)
272 return 0;
273 DEBUG_ATS("_send() %d\r\n", _request.available());
274
275 size_t supply = _request.available();
276 size_t demand = _client->space();
277 if (supply > demand)
278 supply = demand;
279 size_t sent = 0;
280 uint8_t *temp = new uint8_t[100];
281 while (supply)
282 {
283 size_t chunk = supply < 100 ? supply : 100;
284 supply -= _request.read(temp, chunk);
285 sent += _client->add((char *)temp, chunk);
286 }
287 delete temp;
288
289 _client->send();
290 DEBUG_ATS("*sent %d\r\n", sent);
291 _lastActivity = millis();
292 return sent;
293}

324void AsyncTS::_onData(void *Vbuf, size_t len)
325{
326 DEBUG_ATS("_onData handler %.16s... (%d)\r\n", (char *)Vbuf, len);
327 _seize;
328 _lastActivity = millis();
329
330 // Transfer data to xbuf
331
332 _response.write((uint8_t *)Vbuf, len);
333
334 // if header not complete, collect it.
335 // if still not complete, just return.
336
337 while (_state == CONNECTED)
338 {
339 String headerLine = _response.readStringUntil("\r\n");
340
341 // If no line, return.
342
343 if (!headerLine.length())
344 {
345 _lastTSerrorcode = TS_ERR_BAD_RESPONSE;
346 return;
347 }
348
349 // If empty line, all headers are in, advance readyState.
350
351 if (headerLine.length() == 2)
352 {
353 _setState(HEADERSRCVD);
354 }
355
356 // If line is HTTP header, capture HTTPcode.
357
358 else if (headerLine.substring(0, 8) == "HTTP/1.1")
359 {
360 DEBUG_ATS("HTTP/1.1 found.\r\n");
361 _lastTSerrorcode = headerLine.substring(9, headerLine.indexOf(' ', 9)).toInt();
362 DEBUG_ATS("Response code: %d\r\n", _lastTSerrorcode);
363 }
364
365 else if (headerLine.substring(0, 15) == "Content-Length:")
366 {
367 _contentLength = headerLine.substring(16, headerLine.indexOf(' ', 16)).toInt();
368 DEBUG_ATS("Content-Length :%d\r\n", _contentLength);
369 }
370 }
371
372
373 if (_response.available() && _state != RESCOMPLETE)
374 {
375 _setState(RESONGOING);
376 }
377
378 // If not chunked and all data read, close it up.
379
380 if (_response.available() >= _contentLength)
381 {
382 if (_writesession)
383 {
384 if (_response.readString().toInt() == 0)
385 {
386 _lastTSerrorcode = TS_ERR_NOT_INSERTED;
387 }
388 if (_writeResponseUserCB)
389 _writeResponseUserCB(_lastTSerrorcode);
390 }
391 else
392 {
393 if (_retValueSelector)
394 _retValueSelector();
395 }
396 _setState(RESCOMPLETE);
397 _client->stop();
398 _setState(DISCONNECTING);
399 }
400 _release;
401}

978bool AsyncTS::writeFields(unsigned long channelNumber, const char *writeAPIKey)
979{
980 if (!_isReady())
981 {
982 DEBUG_ATS("ats::writeRaw Clinet is busy.");
983 _lastTSerrorcode = TS_ERR_CONNECT_FAILED;
984 return false;
985 }
986 _response.flush();
987 _request.flush();
988 _writesession = true;
989 // Get the content length of the payload
990 int contentLen = _getWriteFieldsContentLength();
991
992 if (contentLen == 0)
993 {
994 // setField was not called before writeFields
995 _lastTSerrorcode = TS_ERR_SETFIELD_NOT_CALLED;
996 if (_writeResponseUserCB)
997 {
998 _writeResponseUserCB(_lastTSerrorcode);
999 }
1000 return false;
1001 }
1002 _request.write("POST /update HTTP/1.1\r\n");
1003 _writeHTTPHeader(writeAPIKey);
1004 _request.write("Content-Type: application/x-www-form-urlencoded\r\n");
1005 _request.write("Content-Length: ");
1006 _request.write(String(contentLen));
1007 _request.write("\r\n\r\n");
1008
1009 bool fFirstItem = true;

1023 }
1024 }

1027 {
1028 if (!fFirstItem)
1029 {
1030 _request.write("&");
1031 }

1038 {
1039 if (!fFirstItem)
1040 {

1042 }

1046 }

1049 {
1050 if (!fFirstItem)
1051 {
1052 _request.write("&");
1053 }
1054 _request.write("elevation=");
1055 _request.write(String(_nextWriteElevation,6));
1056 fFirstItem = false;
1057 }

1060 {
1061 if (!fFirstItem)
1062 {
1063 _request.write("&");
1064 }
1065 _request.write("status=");
1066 _request.write(_nextWriteStatus);
1067 fFirstItem = false;
1068 }

1070 if (_nextWriteTwitter.length() > 0)
1071 {
1072 if (!fFirstItem)
1073 {
1074 _request.write("&");
1075 }
1076 _request.write("twitter=");
1077 _request.write(_nextWriteTwitter);
1078 fFirstItem = false;
1079 }

1082 {
1083 if (!fFirstItem)

1085 _request.write("&");
1086 }
